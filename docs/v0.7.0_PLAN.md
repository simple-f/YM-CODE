```python
# ymcode/core/llm_client.py

class LLMClient:
    """统一 LLM 客户端（兼容现有代码）"""
    
    def __init__(self, config=None):
        self.config = config or {}
        
        # 根据配置选择模型类型
        if self.config.get('use_local'):
            self.model = self._init_local_model()
        else:
            self.model = self._init_api_model()
    
    def _init_local_model(self):
        """初始化本地模型"""
        try:
            from .local_model import LocalModel
            return LocalModel(self.config)
        except ImportError:
            logger.warning("本地模型依赖未安装，回退到 API 模式")
            return self._init_api_model()
    
    def _init_api_model(self):
        """初始化 API 模型"""
        from .api_model import APIModel
        return APIModel(self.config)
    
    def chat(self, prompt, context=None):
        """统一聊天接口（向后兼容）"""
        return self.model.chat(prompt, context)
```

```python
# ymcode/core/local_model.py

class LocalModel:
    """本地模型（新增）"""
    
    def __init__(self, config):
        self.model_path = config.get('local_model_path')
        self.quantization = config.get('quantization', '4bit')
        self.max_tokens = config.get('max_tokens', 8000)
        
        # 延迟加载模型
        self.model = None
    
    def load_model(self):
        """延迟加载模型"""
        if self.model is None:
            # 使用 llama-cpp-python
            from llama_cpp import Llama
            self.model = Llama(
                model_path=self.model_path,
                n_ctx=self.max_tokens,
                n_batch=512
            )
    
    def chat(self, prompt, context=None):
        """聊天接口"""
        self.load_model()
        
        # 构建完整提示
        full_prompt = self._build_prompt(prompt, context)
        
        # 生成响应
        response = self.model(
            full_prompt,
            max_tokens=2000,
            temperature=0.7
        )
        
        return response['choices'][0]['text']
```

```python
# ymcode/core/context_manager.py

class ContextManager:
    """上下文管理器（新增）"""
    
    def __init__(self, max_tokens=8000):
        self.max_tokens = max_tokens
        self.compression_threshold = max_tokens * 0.8
    
    def process(self, context):
        """处理上下文（自动检测和优化）"""
        token_count = self._count_tokens(context)
        
        if token_count > self.max_tokens:
            # 超过限制，需要优化
            return self._optimize_context(context, token_count)
        else:
            # 未超限，直接返回
            return context
    
    def _optimize_context(self, context, token_count):
        """优化上下文"""
        # 1. 尝试压缩
        compressed = self._compress_context(context)
        if self._count_tokens(compressed) <= self.max_tokens:
            return compressed
        
        # 2. 分片处理
        return self._split_context(context)
    
    def _compress_context(self, context):
        """压缩上下文（保留关键信息）"""
        # 移除冗余内容
        # 保留代码和关键对话
        pass
    
    def _split_context(self, context):
        """分片处理"""
        # 将上下文分成多个片段
        chunks = []
        current_chunk = []
        current_tokens = 0
        
        for item in context:
            item_tokens = self._count_tokens(item)
            
            if current_tokens + item_tokens > self.max_tokens:
                # 当前片段已满，开始新片段
                chunks.append(current_chunk)
                current_chunk = [item]
                current_tokens = item_tokens
            else:
                current_chunk.append(item)
                current_tokens += item_tokens
        
        if current_chunk:
            chunks.append(current_chunk)
        
        return chunks
```

```python
# ymcode/skills/code_analyzer.py

from .base import BaseSkill

class CodeAnalyzerSkill(BaseSkill):
    """代码分析技能（新增）"""
    
    def __init__(self):
        super().__init__('code_analyzer')
        self.analyzers = {}
        self._initialized = False
    
    def initialize(self):
        """延迟初始化（按需加载）"""
        if self._initialized:
            return
        
        # 可选依赖，不强制安装
        try:
            import pylint
            from .analyzers.pylint_analyzer import PylintAnalyzer
            self.analyzers['pylint'] = PylintAnalyzer()
        except ImportError:
            pass
        
        try:
            import black
            from .analyzers.black_formatter import BlackFormatter
            self.analyzers['black'] = BlackFormatter()
        except ImportError:
            pass
        
        self._initialized = True
    
    async def execute(self, arguments):
        """执行代码分析"""
        self.initialize()  # 按需加载
        
        code = arguments.get('code')
        language = arguments.get('language', 'python')
        
        results = []
        for name, analyzer in self.analyzers.items():
            if analyzer.supports(language):
                try:
                    result = analyzer.analyze(code)
                    results.append({
                        'tool': name,
                        'result': result
                    })
                except Exception as e:
                    logger.error(f"{name} 分析失败：{e}")
        
        return {
            'success': True,
            'results': results,
            'analyzers_used': list(self.analyzers.keys())
        }
```

#### 2. 分析器实现

```python
# ymcode/skills/analyzers/pylint_analyzer.py

class PylintAnalyzer:
    """Pylint 代码分析器"""
    
    def supports(self, language):
        return language == 'python'
    
    def analyze(self, code):
        """分析代码"""
        import tempfile
        import subprocess
        
        # 创建临时文件
        with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
            f.write(code)
            temp_file = f.name
        
        # 运行 Pylint
        result = subprocess.run(
            ['pylint', temp_file],
            capture_output=True,
            text=True
        )
        
        # 解析结果
        issues = self._parse_pylint_output(result.stdout)
        
        return {
            'issues': issues,
            'score': self._extract_score(result.stdout)
        }
```
